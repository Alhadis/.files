(Re)generate this list using the following shell-script:

#!/bin/sh
set -e

# List pre-10.6 voices with phoneme directive support (requires `sox`)
for v in `say -v '?' | sed 's/[[:blank:]].*//'`; do
say -v "$v" \
	--file-format=WAVE \
	--data-format=LEF32@32000 \
	-o /tmp/voice-test.wav \
	'[[inpt PHON]]@@@'
case `sox /tmp/voice-test.wav -n stat 2>&1 | sed -n '
	/^Maximum amplitude:/! d
	s/.*: *\([^ ]*\)$/6k0[1+]sR0 0 \1!>Rn/p
' | dc` in 1) echo "$v";; esac
done || :; unset v
rm -f /tmp/voice-test.wav
